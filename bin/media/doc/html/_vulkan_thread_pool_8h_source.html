36  // Loop through all remaining jobs
37  void queueLoop()
38  {
39  while (true)
40  {
41  std::function<void()> job;
42  {
43  std::unique_lock<std::mutex> lock(queueMutex);
44  condition.wait(lock, [this] { return !jobQueue.empty() || destroying; });
45  if (destroying)
46  {
47  break;
48  }
49  job = jobQueue.front();
50  }
51 
52  job();
53 
54  {
55  std::lock_guard<std::mutex> lock(queueMutex);
56  jobQueue.pop();
57  condition.notify_one();
58  }
59  }
60  }
